1、ES5

```js
    function selectSort(arr) {
      let index

      for (let i = 0; i < arr.length; i++) {
        index = i

        for (let j = i + 1; j < arr.length; j++) {
          if (arr[j] < arr[index]) {
            index = j
          }
        }

        if (index !== i) {
          // 交换位置
          var temp = arr[i];
          arr[i] = arr[index];
          arr[index] = temp;
        }
      }

      return arr
    }
```

2、ES6 解构赋值

```js
    function selectSort(arr) {
      let index

      for (let i = 0; i < arr.length; i++) {
        index = i

        for (let j = i + 1; j < arr.length; j++) {
          if (arr[j] < arr[index]) {
            index = j
          }
        }

        if (index !== i) {
          [arr[i], arr[index]] = [arr[index], arr[i]]
        }
      }

      return arr
    }
```
